        function toggleDisease(key) {

            if (selectedDiseases.has(key)) {

                selectedDiseases.delete(key);

            } else {

                selectedDiseases.add(key);

            }

            updateUI();

        }


        function updateUI() {

            // 更新复选框视觉状态

            document.querySelectorAll('.checkbox-wrapper').forEach(cb => {

                const dot = cb.nextElementSibling.querySelector('.check-dot');

                const circle = cb.nextElementSibling.querySelector('.check-circle');

                if (cb.checked) {

                    dot.classList.remove('hidden');

                    circle.classList.add('border-blue-600');

                } else {

                    dot.classList.add('hidden');

                    circle.classList.remove('border-blue-600');

                }

            });


            if (selectedDiseases.size === 0) {

                document.getElementById('resultsArea').classList.add('hidden');

            }

            document.getElementById('resultsArea').classList.remove('hidden');

        }

        function copyToClipboard() {

            let text = `脊柱外科检查医嘱清单\n生成时间：${new Date().toLocaleString()}\n\n`;

            

            selectedDiseases.forEach(key => {

                const data = database[key];

                text += `【${data.name}】\n`;

                text += `检验项目：\n${data.检验.map(i => '- ' + i.replace(/<[^>]*>/g, '')).join('\n')}\n\n`;

                text += `检查项目：\n${data.检查.map(i => '- ' + i.replace(/<[^>]*>/g, '')).join('\n')}\n\n`;

            });


            navigator.clipboard.writeText(text).then(() => {

                alert('已复制到剪贴板！可直接粘贴到病程记录或医嘱本。');

            });

        }
